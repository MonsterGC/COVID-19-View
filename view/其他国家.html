filter_key = data.groupby('国家/地区').mean()["累计死亡人数"].sort_values(ascending=False)[0:8]

for tmp in provinces_data:
    if tmp in filter_key:
        x = range(0, len(provinces_data[tmp]['累计死亡人数']))
#     label = lambda label:'' if tmp not in filter_key else tmp
        plt.plot(x, provinces_data[tmp]['累计死亡人数'] ,  mec='r', mfc='w', label= tmp)
plt.legend()  # 让图例生效
plt.xlabel('时间(/天)') #X轴标签
plt.ylabel("累计死亡人数-均值最大前8") #Y轴标签
plt.show()
