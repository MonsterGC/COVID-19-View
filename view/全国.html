squares=data['确诊(新增）']
input_value=range(len(squares))
#plt.plot(squares)
plt.plot(input_value,squares,linewidth=1)
 
#设置图表标题，并给坐标轴加标签

plt.tick_params(axis='both',labelsize=14)
plt.show()
